💻 Como os Produtos São Criados no Código

// CLASSE ABSTRATA
public abstract class Produto {
    protected String nome;
    protected double preco;
    
    public abstract void exibirDetalhes();
}

// CLASSE CONCRETA
public class Camisa extends Produto {
    private String tamanho;
    
    public Camisa(String nome, double preco, String tamanho) {
        super(nome, preco);
        this.tamanho = tamanho;
    }
    
    @Override
    public void exibirDetalhes() {
        System.out.println("👕 " + nome + " - Tamanho: " + tamanho 
                         + " - R$ " + preco);
    }
}

// USO
Camisa camisa1 = new Camisa("Camisa Azul", 89.90, "M");
camisa1.exibirDetalhes(); // Output: 👕 Camisa Azul - Tamanho: M - R$ 89.90
