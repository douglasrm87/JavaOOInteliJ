💻 Como Calcular o Total com Streams

// CLASSE: Carrinho
public class Carrinho {
    private List<Produto> itens;
    
    // Método que usa STREAMS para calcular total
    public double calcularTotal() {
        return itens.stream()                    // Converte lista em stream
                    .mapToDouble(Produto::getPreco)  // Extrai preço (double)
                    .sum();                      // Soma todos os preços
    }
}

// FORMA ANTIGA (sem Streams)
public double calcularTotalAntigo() {
    double total = 0;
    for (Produto p : itens) {
        total += p.getPreco();
    }
    return total;
}

// ✅ Streams é mais elegante, funcional e legível!
